import 'package:flutter/material.dart';
import 'package:firebase_core/firebase_core.dart';
import 'package:sponsorenlauf_app/pages/public_landing_page.dart';
import 'firebase_options.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp(
    options: DefaultFirebaseOptions.currentPlatform,
  );
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Sponsorenlauf App',
      theme: ThemeData(
        primaryColor: const Color(0xFFFFD200),
        colorScheme: ColorScheme.fromSwatch().copyWith(
          primary: const Color(0xFFFFD200),
          secondary: const Color(0xFF005DA4),
        ),
        scaffoldBackgroundColor: Colors.grey[100],
        appBarTheme: const AppBarTheme(
          backgroundColor: Color(0xFFFFD200),
          foregroundColor: Color(0xFF005DA4),
          centerTitle: true,
        ),
      ),
      // Wir verwenden die `home`-Eigenschaft für einen stabilen Startpunkt.
      // Benannte Routen hatten in der alten IDE zu Instabilitäten geführt.
      home: const PublicLandingPage(),
    );
  }
}
